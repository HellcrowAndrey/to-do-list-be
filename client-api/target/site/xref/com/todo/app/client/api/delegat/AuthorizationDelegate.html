46      public List<Task> getData(@Valid UserModel authUser) {
47          if (authUser == null) {
48              logger.warn("User model is not valid.", AuthorizationDelegate.class);
49              return null;
50          }
51          CacheManager manager = CacheManager.getInstance();
52          UserModel userInCache = manager.fetchUser(authUser);
53          if (userInCache != null) {
54              logger.info("User find in cache.", AuthorizationDelegate.class);
55              return manager.fetchTasks(userInCache);
56          }
57          return findData(authUser);
58      }
59

66              return null;
67          } else {
68              CacheManager manager = CacheManager.getInstance();
69              logger.info("User find in db and add to cache.", AuthorizationDelegate.class);
70              manager.addUser(userInDb);
71              IServiceTasks serviceTasks = new ServiceTasksImpl();
72              List<Task> tasks = serviceTasks.read(userInDb.getLogin());
73              logger.info("Return Tasks by user.", AuthorizationDelegate.class);
74              return manager.addTasks(tasks, userInDb.getLogin());
75          }
